screen_capture.py -> exports:
  capture_screen(region: dict | None = None) -> np.ndarray
    # Captures screen or a region. Returns BGR numpy array (uint8, HxWxC).
    # region format: {"top": int, "left": int, "width": int, "height": int}
    # If region is None, captures full primary monitor (mss.monitors[1]).
    # Uses context manager — mss handle is released after every call.

# =============================================================================
# vision.py
# Role: Analyzes screen frames. Returns structured description for the agent.
# Imports from: screen_capture.py, cv2, easyocr, numpy
# Status: Complete
# FRAGILE: _reader (easyocr.Reader) is initialized at module load with gpu=True.
#          Do NOT move it inside analyze_frame() — re-init on every call = 2-5s penalty per tick.
#          analyze_frame() input MUST be BGR np.ndarray. BGRA or PIL.Image will break the pipeline.
#          Return dict keys are exactly: "description", "elements", "text".
#          agent_loop.py keys into these by name — do NOT rename them.
# NOTE: runtime_models.py exists as the future typed baseline, but vision.py still
#       returns the legacy dict contract until all callers are migrated together.
# =============================================================================

vision.py -> exports:
  analyze_frame(img: np.ndarray) -> dict
    # Analyzes a BGR screen frame using EasyOCR + OpenCV contour detection.
    # Two-pass: OCR pass extracts text_block elements; contour pass finds button_candidates.
    # Return format: {"description": str, "elements": list, "text": str}
    # Each element: {"type": str, "bbox": [x, y, w, h], "text": str, "confidence": float}
    # Element types: "text_block", "button_candidate"
    # description is capped at 300-char text preview to avoid bloating agent prompt context.

  get_screen_state(region: dict | None = None) -> dict
    # Convenience wrapper: calls capture_screen(region) then analyze_frame(img).
    # No direct mss usage inside this function.
    # Returns same format as analyze_frame():
    # {"description": str, "elements": list, "text": str}

# ollama_client.py
# Role: Interface to local Ollama model via HTTP API.
# Imports from: config.py (MODEL_NAME, OLLAMA_URL, MAX_RETRIES, DEBUG), requests
# Status: Complete
# FRAGILE: query_model() uses /api/chat with stream:False. Return type is str.
#          Do NOT switch to /api/generate or enable streaming.
#          load_model() uses /api/generate with empty prompt + keep_alive — correct warm-up pattern.
#          check_ollama_running() guard in query_model() is what raises ConnectionError. Do not remove.
# NOTE: runtime_models.py defines the future ModelReply baseline, but ollama_client.py
#       still returns str until the backend-adapter migration lands.
# =============================================================================

ollama_client.py -> exports:
  check_ollama_running() -> bool
    # GET to OLLAMA_URL root with 3s timeout.
    # Returns True if HTTP 200, False on any exception.
    # Called by query_model() as entry guard.

  load_model(model_name: str) -> bool
    # Warms up / preloads a model in Ollama via POST /api/generate.
    # Payload: empty prompt + keep_alive: "5m". Timeout: 30s.
    # Returns True if successful (HTTP 200), False if failed.

  query_model(prompt: str, system: str | None = None) -> str
    # Sends prompt to Ollama model via POST /api/chat with stream:False.
    # Uses MODEL_NAME and OLLAMA_URL from config.py.
    # Optional system param prepended as system message if provided.
    # Retries up to MAX_RETRIES times on transient failure.
    # Raises: ConnectionError if Ollama not running or all retries exhausted.

# agent_loop.py
# Role: Core autonomous goal loop. Orchestrates all components.
# Imports from: config.py (MODEL_NAME, OLLAMA_URL, LOOP_DELAY, DEBUG, BLOCK_CPU_COMPUTE, OLLAMA_NUM_GPU)
#               vision.py (get_screen_state)
#               ollama_client.py (check_ollama_running, query_model)
#               open-interpreter (interpreter)
#               time (stdlib)
#               os (stdlib)
# Status: Complete
# FRAGILE: interpreter is configured at module load — do NOT re-configure in main.py or elsewhere.
#          run() resets interpreter.messages = [] at start — each goal gets a clean session.
#          GOAL_COMPLETE is the exact completion signal string — do not change it.
#          _running is module-level state shared between run() and stop() — do not make it local.
#          auto_run=True means OI executes code without confirmation. Intentional for autonomy.
#          offline=True prevents cloud fallback. Do not remove.
#          CUDA environment variables set at module load BEFORE interpreter init — do not reorder.
#          CUDA_VISIBLE_DEVICES=0 hardcoded to GPU 0 (RTX 4070) — multi-GPU needs adjustment.
#          num_gpu=999 is intentional — signals unlimited GPU layers with RAM offload.
# NOTE: agent_loop.py still uses the legacy single-step loop. Queue migration has NOT started yet.
# =============================================================================

agent_loop.py -> exports:
  run(goal: str) -> None
    # Main entry point for the agent loop.
    # Calls check_ollama_running() at entry — raises ConnectionError if not live.
    # Resets interpreter session (messages = [], system_message = _SYSTEM_PROMPT).
    # Loop: get_screen_state() -> step() -> check GOAL_COMPLETE -> sleep(LOOP_DELAY).
    # Exits on GOAL_COMPLETE in action string, KeyboardInterrupt, or stop() call.

  step(goal: str, screen_state: dict) -> str
    # Runs a single agent tick.
    # Builds prompt from goal + screen_state["description"] + screen_state["text"][:500].
    # Routes through interpreter.chat(display=DEBUG, stream=False).
    # Returns last assistant message content as action string.

  stop() -> None
    # Sets _running = False — run() exits on next loop iteration.
    # Clears interpreter.messages to free memory.

# =============================================================================
# main.py
# Role: Entry point. Wires config + agent loop together. Minimal logic.
# Imports from: config.py (DEBUG, MODEL_NAME, OLLAMA_URL)
#               agent_loop.py (run)
#               ollama_client.py (check_ollama_running, load_model) [startup only — not runtime]
# Status: Complete
# NOTE: ollama_client is imported ONLY for the startup Ollama check and model warm-up.
#       All runtime Ollama calls go through agent_loop.py.
#       main.py does NOT import vision.py or configure the interpreter.
# =============================================================================

main.py -> exports:
  main() -> None
    # 1. Prints startup banner (model, URL, debug state).
    # 2. Calls check_ollama_running() — exits with sys.exit(1) if Ollama not live.
    # 3. Calls load_model(MODEL_NAME) — warns but continues if warm-up fails.
    # 4. Prompts user for goal string via input(). Exits cleanly on empty or Ctrl+C.
    # 5. Calls agent_loop.run(goal).
    # 6. Catches ConnectionError and unexpected exceptions — exits with message.
    #    DEBUG=True re-raises unexpected exceptions for full traceback.
